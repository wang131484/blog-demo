<template>
  <div id="cart">
    <headNav NavTitle="购物车" class="head"/>
    <div class="common">
      <van-checkbox-group v-model="checkedArr">
        <van-checkbox
          v-for="(item, index) in dataone"
          :key="index"
          :label-disabled="true"
          :name="item.id"
          class="lists"
          checked-color="#fc8fb4"
        >
          <!-- @click="check(item.id)" -->
          <img :src="item.goods.image" alt>
          <div class="liststext">
            <div class="comtop">
              <p>{{item.goods.title}}</p>
              <van-icon name="delete" size="18" @click="del=item.id;tips=true;"/>
            </div>
            <div>
              <div class="font1">
                <span>￥</span>
                <span>{{item.goods.price}}</span>
              </div>
              <div>
                <van-stepper
                  disable-input
                  v-model="item.num"
                  @plus="number(item.id,item.num+1)"
                  @minus="number(item.id,item.num-1)"
                />
              </div>
            </div>
          </div>
        </van-checkbox>
      </van-checkbox-group>
    </div>
    <van-submit-bar
      :price="getTotal"
      button-text="结算"
      button-type="danger"
      class="sum"
      @submit="sub"
    >
      <van-checkbox
        style="margin-left:13px;"
        checked-color="#FC8FB4"
        v-model="isAllchecked"
        @click="allchecked"
      >全选</van-checkbox>
    </van-submit-bar>
    <van-popup v-model="tips">
      <div class="tips">
        <div class="tips_del">确定删除该商品吗</div>
        <div class="tips_p">
          <div class="p_quxiao" @click="tips = false">取消</div>

          <div class="p_queding" @click="handleRemove">确定</div>
        </div>
      </div>
    </van-popup>
  </div>
</template>
<script>
// import { log } from 'util';
export default {
  data() {
    return {
      tips: false, //删除弹窗是否出现
      dataone: [], //购物车数据
      checkedArr: [], //选中的商品ID集合
      del: "", //购物车id
      money: 0 //总金额
    };
  },
  created() {
    this.cart(); //购物车调用接口
  },
  methods: {
    //获取商品列表
    cart() {
      this.$httpRequest("/app/cart", "GET").then(res => {
        console.log("购物车列表", res);
        this.dataone = res.data.data;
      });
    },
    //删除商品
    handleRemove() {
      this.tips = false;
      this.$httpRequest("/app/cart", "DELETE", { id: this.del }).then(res => {
        if (res.code == 1) {
          this.$toast("删除成功");
          this.cart();
        }
      });
    },
    // 修改购物车数量
    number(id, sum) {
      this.$httpRequest("/app/cart", "PUT", {
        id: id,
        num: sum
      }).then(res => {
        console.log(res);
      });
    },
    //全选
    allchecked() {
      //如果未全选
      if (!this.isAllchecked) {
        for (let i = 0; i < this.dataone.length; i++) {
          if (this.checkedArr.indexOf(this.dataone[i].id) == -1) {
            this.checkedArr.push(this.dataone[i].id);
          }
        }
      } else {
        //如果全选
        this.checkedArr = [];
      }
      this.isAllchecked = !this.isAllchecked;
    },
    //结算
    sub() {
      this.$router.push({
        name: "placeOrder",
        query: {
          carts: this.checkedArr,
          type: 2
        }
      });
    }
  },
  computed: {
    //计算总价
    getTotal() {
      var total = 0;
      for (let i = 0; i < this.checkedArr.length; i++) {
        for (let a = 0; a < this.dataone.length; a++) {
          if (this.checkedArr[i] == this.dataone[a].id) {
            total += this.dataone[a].num * this.dataone[a].goods.price;
          }
        }
      }
      return total * 100;
    },
    //是否全选
    isAllchecked() {
      for (let i = 0; i < this.dataone.length; i++) {
        if (this.checkedArr.indexOf(this.dataone[i].id) == -1) {
          return false;
        }
      }
      return true;
    }
  }
};
</script>

<script>
export default {
  data() {
    return {
      ent_type: this.$route.query.type, //页面进入方式 1-商城 2-购物车
      ent_id: this.$route.query.id, //页面进入id
      carts: this.$route.query.carts, //购物车主键Id
      list_num: 1, //商品数量
      order_list: [], //数据 商城
      lists: "", //商城商品详情
      order_car_list: [], //数据 购物车
      cars:'',//购物车商品详情
      address_id: "", //地址id
      goods_id: "" //订单id
    };
  },
  created() {
    //获取订单信息
    this.get_user();
  },
  methods: {
    //获取订单信息 商城
    get_user() {
      this.$toast.loading({
        duration: 0,
        message: "加载中..."
      });
      if (this.ent_type == 1) {
        console.log("商城");
        this.$httpRequest("/app/order/order_fast", "POST", {
          goods_id: this.ent_id,
          counts: 1
        }).then(res => {
          console.log("获取商城订单信息", res);
          if (res.code == 1) {
            this.$toast.clear();
            this.order_list = res.data;
            this.lists = res.data.goods;
            this.address_id = res.data.address.id;
            this.goods_id = res.data.goods.id;
          }
        });
      } else {
        console.log("购物车");
        this.$httpRequest("/app/order/order_details", "POST", {
          carts: this.carts
        }).then(res => {
          console.log("获取购物车订单信息", res);
          if (res.code == 1) {
            this.$toast.clear();
            this.order_car_list = res.data;
            this.cars = res.data.goods
            this.address_id = res.data.address.id;
            this.goods_id = res.data.goods.id;
          }
        });
      }
    },
    //地址
    nav_add() {
      this.$router.push({
        name: "address"
      });
    },
    //获取数量
    onChange(value) {
      console.log(value);
    },
    // 提交订单
    get_order() {
      this.$toast.loading({
        duration: 0,
        message: "加载中..."
      });
      if (this.ent_type == 1) {
        console.log("商城");
        this.$httpRequest("/app/order/order_generate", "POST", {
          address_id: this.address_id,
          goods_id: this.goods_id,
          counts: this.list_num
        }).then(res => {
          console.log("订单信息id", res);
          if (res.code == 1) {
            this.$toast.clear();
            this.$router.push({
              name: "checkout",
              query: {
                order_id: res.data.id,
                id: this.ent_id,
                list_num: this.list_num
              }
            });
          } else {
            this.$toast({
              message: res.msg,
              duration: 2000
            });
          }
        });
      } else {
        console.log("购物车");
        this.$httpRequest("/app/order", "POST", {
          address_id: this.address_id,
          carts: this.carts
        }).then(res => {
          console.log("订单信息id", res);
          if (res.code == 1) {
            this.$toast.clear();
            this.$router.push({
              name: "checkout"
              // query: {
              //   order_id: res.data.id,
              //   id: this.ent_id,
              //   list_num: this.list_num
              // }
            });
          } else {
            this.$toast({
              message: res.msg,
              duration: 2000
            });
          }
        });
      }
    }
  },
  computed:{
    getTotal() {
      var total = 0;
      for (let i = 0; i < this.cars.length; i++) {
        total +=
          (this.cars[i].goods.price * this.cars[i].num*this.order_car_list.discount)/10;
      }
      return total;
    }
  }
};
</script>
