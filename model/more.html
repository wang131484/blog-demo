      </div>
      <div class="tab-item" @click="switchTo('shop')">
        <img
          :src="'shop' === $route.name ? tabBarImgArr[1].selected : tabBarImgArr[1].normal"

        >
        <span :class="{on: 'shop' === $route.name}">商城</span>
      </div>
      <div class="tab-item" @click="switchTo('PersonalCenter')">
        <img
          :src="'PersonalCenter' === $route.name ? tabBarImgArr[2].selected : tabBarImgArr[2].normal"

      </div>
    </div>
  </div>
</template>

<script>
export default {
  components: {},
  data() {
    return {
      tabBarImgArr: [

};
</script>
